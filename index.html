function doGet() { var sheet = SpreadsheetApp.getActiveSpreadsheet().getActiveSheet(); var wird = sheet.getRange("B1").getValue(); // الورد في B1 var names = sheet.getRange("A:A").getValues().filter(String).flat(); // الأسماء في A return ContentService.createTextOutput(JSON.stringify({ "wirdText": wird, "leaderboard": names })).setMimeType(ContentService.MimeType.JSON); } function doPost(e) { var sheet = SpreadsheetApp.getActiveSpreadsheet().getActiveSheet(); var name = e.parameter.name; if (name) { sheet.appendRow([name]); } return ContentService.createTextOutput("Success"); }
